# Data info

- aspiration -> type of aspiration used in a car
- std = standard
- turbo = turbo
- enginelocation -> location of the engine on the car
- carwidth -> width of the car
- curbweight -> weight of the car
- enginetype -> type of engine on the car
  - dohc = dual overhead cam
  - dohcv = dual overhead cam and valve
  - l = L engine
  - ohc = overhead cam
  - ohcf = overhead cam and valve F engine
  - ohcv = overhead cam and valve
  - rotor = rotary engine
- cylindernumber -> number of cylinders in the car
- stroke -> distance traveled by the piston for every half-turn (=
stroke)
